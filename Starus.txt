// helper function
const formatJsonToMarkdown = (obj, indent = 0) => {
  let md = "";
  const spaces = " ".repeat(indent * 2);

  for (let key in obj) {
    if (typeof obj[key] === "object" && obj[key] !== null) {
      md += `${spaces}${key}:\n`;
      md += formatJsonToMarkdown(obj[key], indent + 1);
    } else {
      md += `${spaces}${key}: ${obj[key]}\n`;
    }
  }

  return md;
};
