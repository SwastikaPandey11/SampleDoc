// src/pages/LoginPage.jsx
import { useState } from "react";
import { validateLogin } from "../utils/auth";
import { useNavigate } from "react-router-dom";
import { Box, TextField, Button, Typography } from "@mui/material";

export default function LoginPage() {
  const [user, setUser] = useState("");
  const [pass, setPass] = useState("");
  const [error, setError] = useState("");
  const navigate = useNavigate();

  const handleLogin = () => {
    const valid = validateLogin(user, pass);
    if (!valid) {
      setError("Invalid credentials");
      return;
    }
    sessionStorage.setItem("bank", valid.bank);
    sessionStorage.setItem("loginTime", Date.now());
    navigate("/dashboard");
  };

  return (
    <Box sx={{ display: "flex", flexDirection: "column", gap: 2, mt: 8, mx: "auto", width: 300 }}>
      <Typography variant="h5" align="center">FinBridgeAI Login</Typography>
      <TextField label="Username" value={user} onChange={e => setUser(e.target.value)} />
      <TextField label="Password" type="password" value={pass} onChange={e => setPass(e.target.value)} />
      {error && <Typography color="error">{error}</Typography>}
      <Button variant="contained" onClick={handleLogin}>Login</Button>
    </Box>
  );
}
